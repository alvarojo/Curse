//Set the icon item
setIcon("openblocks:tasty_clay")

//Set the title String
setTitle("TASTE THE CLAY")

//Set the description String
setDescription("openblocks, memeing way before the meme")

//Set the parent advancement. The syntax is "modid:folder/path/to/the/advancement/then/filename"
setParent("triumph:example/tablecraft")


criteria = addCriteria("consumed", "minecraft:consume_item")

//Sets the required item for the criteria. This is the item that needs to be consumed. First variable is the item name, second is the metadata and is not required.
criteria.setItem("openblocks:tasty_clay")


//Sets the item reward. A second variable can be added for meta, and a third can be added for stack size
setRewardExperience(69)

//By not setting anything about requirements here, all criteria are automatically required
